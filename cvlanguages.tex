%%%% Select one of following languages
% ADDHERE new language to pass to babel
\DeclareOption{english}{
	\PassOptionsToPackage{\CurrentOption}{babel}
}
\DeclareOption{czech}{
	\PassOptionsToPackage{\CurrentOption}{babel}
}

% Fallback
\DeclareOption*{
	\ClassError{cv}{Unknown option '\CurrentOption'}{}
}

% Process given options
\ProcessOptions\relax

%%%% Check if language was set
% ADDHERE new language to check if language was set
\newboolean{cvlanguageset}   
\setboolean{cvlanguageset}{false}

% Check all possible languages here
\iflanguage{english}{
	\setboolean{cvlanguageset}{true}
}{}
\iflanguage{czech}{
	\setboolean{cvlanguageset}{true}

}{}

% Check result
\ifthenelse{\boolean{cvlanguageset}}{
	% OK, language was set
}{
	\ClassError{cv}{Language was not set!}{}
}

%%%% Define language filter command
% ADDHERE new language filter
\newcommand{\cveng}[1]{%
	\iflanguage{english}{#1}{}%
}
\newcommand{\cvcze}[1]{%
	\iflanguage{czech}{#1}{}%
}
